113 unsigned char asc2nibble(char c) {
114 
115  if ((c >= '0') && (c <= '9'))
116  return c - '0';
117 
118  if ((c >= 'A') && (c <= 'F'))
119  return c - 'A' + 10;
120 
121  if ((c >= 'a') && (c <= 'f'))
122  return c - 'a' + 10;
123 
124  return 16; /* error */
125 }

127 int hexstring2data(char *arg, unsigned char *data, int maxdlen) {
128 
129  int len = strlen(arg);
130  int i;
131  unsigned char tmp;
132 
133  if (!len || len%2 || len > maxdlen*2)
134  return 1;
135 
136  memset(data, 0, maxdlen);
137 
138  for (i=0; i < len/2; i++) {
139 
140  tmp = asc2nibble(*(arg+(2*i)));
141  if (tmp > 0x0F)
142  return 1;
143 
144  data[i] = (tmp << 4);
145 
146  tmp = asc2nibble(*(arg+(2*i)+1));
147  if (tmp > 0x0F)
148  return 1;
149 
150  data[i] |= tmp;
151  }
152 
153  return 0;
154 }

156 int parse_canframe(char *cs, struct canfd_frame *cf) {
157  /* documentation see lib.h */
158 
159  int i, idx, dlen, len;
160  int maxdlen = CAN_MAX_DLEN;
161  int ret = CAN_MTU;
162  canid_t tmp;
163 
164  len = strlen(cs);
165  //printf("'%s' len %d\n", cs, len);
166 
167  memset(cf, 0, sizeof(*cf)); /* init CAN FD frame, e.g. LEN = 0 */
168 
169  if (len < 4)
170  return 0;
171 
172  if (cs[3] == CANID_DELIM) { /* 3 digits */
173 
174  idx = 4;
175  for (i=0; i<3; i++){
176  if ((tmp = asc2nibble(cs[i])) > 0x0F)
177  return 0;
178  cf->can_id |= (tmp << (2-i)*4);
179  }
180 
181  } else if (cs[8] == CANID_DELIM) { /* 8 digits */
182 
183  idx = 9;
184  for (i=0; i<8; i++){
185  if ((tmp = asc2nibble(cs[i])) > 0x0F)
186  return 0;
187  cf->can_id |= (tmp << (7-i)*4);
188  }
189  if (!(cf->can_id & CAN_ERR_FLAG)) /* 8 digits but no errorframe? */
190  cf->can_id |= CAN_EFF_FLAG; /* then it is an extended frame */
191 
192  } else
193  return 0;
194 
195  if((cs[idx] == 'R') || (cs[idx] == 'r')){ /* RTR frame */
196  cf->can_id |= CAN_RTR_FLAG;
197 
198  /* check for optional DLC value for CAN 2.0B frames */
199  if(cs[++idx] && (tmp = asc2nibble(cs[idx++])) <= CAN_MAX_DLEN) {
200  cf->len = tmp;
201 
202  /* check for optional raw DLC value for CAN 2.0B frames */
203  if ((tmp == CAN_MAX_DLEN) && (cs[idx++] == CC_DLC_DELIM)) {
204  tmp = asc2nibble(cs[idx]);
205  if ((tmp > CAN_MAX_DLEN) && (tmp <= CAN_MAX_DLC)) {
206  struct can_frame *ccf = (struct can_frame *)cf;
207 
208  ccf->can_dlc = tmp;
209  }
210  }
211  }
212  return ret;
213  }
214 
215  if (cs[idx] == CANID_DELIM) { /* CAN FD frame escape char '##' */
216 
217  maxdlen = CANFD_MAX_DLEN;
218  ret = CANFD_MTU;
219 
220  /* CAN FD frame <canid>##<flags><data>* */
221  if ((tmp = asc2nibble(cs[idx+1])) > 0x0F)
222  return 0;
223 
224  cf->flags = tmp;
225  idx += 2;
226  }
227 
228  for (i=0, dlen=0; i < maxdlen; i++){
229 
230  if(cs[idx] == DATA_SEPERATOR) /* skip (optional) separator */
231  idx++;
232 
233  if(idx >= len) /* end of string => end of data */
234  break;
235 
236  if ((tmp = asc2nibble(cs[idx++])) > 0x0F)
237  return 0;
238  cf->data[i] = (tmp << 4);
239  if ((tmp = asc2nibble(cs[idx++])) > 0x0F)
240  return 0;
241  cf->data[i] |= tmp;
242  dlen++;
243  }
244  cf->len = dlen;
245 
246  /* check for extra DLC when having a Classic CAN with 8 bytes payload */
247  if ((maxdlen == CAN_MAX_DLEN) && (dlen == CAN_MAX_DLEN) && (cs[idx++] == CC_DLC_DELIM)) {
248  unsigned char dlc = asc2nibble(cs[idx]);
249 
250  if ((dlc > CAN_MAX_DLEN) && (dlc <= CAN_MAX_DLC)) {
251  struct can_frame *ccf = (struct can_frame *)cf;
252 
253  ccf->can_dlc = dlc;
254  }
255  }
256 
257  return ret;
258 }

271 void sprint_canframe(char *buf , struct canfd_frame *cf, int sep, int maxdlen) {
272  /* documentation see lib.h */
273 
274  int i,offset;
275  int len = (cf->len > maxdlen) ? maxdlen : cf->len;
276 
277  if (cf->can_id & CAN_ERR_FLAG) {
278  put_eff_id(buf, cf->can_id & (CAN_ERR_MASK|CAN_ERR_FLAG));
279  buf[8] = '#';
280  offset = 9;
281  } else if (cf->can_id & CAN_EFF_FLAG) {
282  put_eff_id(buf, cf->can_id & CAN_EFF_MASK);
283  buf[8] = '#';
284  offset = 9;
285  } else {
286  put_sff_id(buf, cf->can_id & CAN_SFF_MASK);
287  buf[3] = '#';
288  offset = 4;
289  }
290 
291  /* standard CAN frames may have RTR enabled. There are no ERR frames with RTR */
292  if (maxdlen == CAN_MAX_DLEN && cf->can_id & CAN_RTR_FLAG) {
293  buf[offset++] = 'R';
294  /* print a given CAN 2.0B DLC if it's not zero */
295  if (cf->len && cf->len <= CAN_MAX_DLEN) {
296  buf[offset++] = hex_asc_upper_lo(cf->len);
297 
298  /* check for optional raw DLC value for CAN 2.0B frames */
299  if (cf->len == CAN_MAX_DLEN) {
300  struct can_frame *ccf = (struct can_frame *)cf;
301 
302  if ((ccf->can_dlc > CAN_MAX_DLEN) && (ccf->can_dlc <= CAN_MAX_DLC)) {
303  buf[offset++] = CC_DLC_DELIM;
304  buf[offset++] = hex_asc_upper_lo(ccf->can_dlc);
305  }
306  }
307  }
308 
309  buf[offset] = 0;
310  return;
311  }
312 
313  if (maxdlen == CANFD_MAX_DLEN) {
314  /* add CAN FD specific escape char and flags */
315  buf[offset++] = '#';
316  buf[offset++] = hex_asc_upper_lo(cf->flags);
317  if (sep && len)
318  buf[offset++] = '.';
319  }
320 
321  for (i = 0; i < len; i++) {
322  put_hex_byte(buf + offset, cf->data[i]);
323  offset += 2;
324  if (sep && (i+1 < len))
325  buf[offset++] = '.';
326  }
327 
328  /* check for extra DLC when having a Classic CAN with 8 bytes payload */
329  if ((maxdlen == CAN_MAX_DLEN) && (len == CAN_MAX_DLEN)) {
330  struct can_frame *ccf = (struct can_frame *)cf;
331  unsigned char dlc = ccf->can_dlc;
332 
333  if ((dlc > CAN_MAX_DLEN) && (dlc <= CAN_MAX_DLC)) {
334  buf[offset++] = CC_DLC_DELIM;
335  buf[offset++] = hex_asc_upper_lo(dlc);
336  }
337  }
338 
339  buf[offset] = 0;
340 }

357 void sprint_long_canframe(char *buf , struct canfd_frame *cf, int view, int maxdlen) {
358  /* documentation see lib.h */
359 
360  int i, j, dlen, offset;
361  int len = (cf->len > maxdlen)? maxdlen : cf->len;
362 
363  /* initialize space for CAN-ID and length information */
364  memset(buf, ' ', 15);
365 
366  if (cf->can_id & CAN_ERR_FLAG) {
367  put_eff_id(buf, cf->can_id & (CAN_ERR_MASK|CAN_ERR_FLAG));
368  offset = 10;
369  } else if (cf->can_id & CAN_EFF_FLAG) {
370  put_eff_id(buf, cf->can_id & CAN_EFF_MASK);
371  offset = 10;
372  } else {
373  if (view & CANLIB_VIEW_INDENT_SFF) {
374  put_sff_id(buf + 5, cf->can_id & CAN_SFF_MASK);
375  offset = 10;
376  } else {
377  put_sff_id(buf, cf->can_id & CAN_SFF_MASK);
378  offset = 5;
379  }
380  }
381 
382  /* The len value is sanitized by maxdlen (see above) */
383  if (maxdlen == CAN_MAX_DLEN) {
384  if (view & CANLIB_VIEW_LEN8_DLC) {
385  struct can_frame *ccf = (struct can_frame *)cf;
386  unsigned char dlc = ccf->can_dlc;
387 
388  /* fall back to len if we don't have a valid DLC > 8 */
389  if (!((len == CAN_MAX_DLEN) && (dlc > CAN_MAX_DLEN) &&
390  (dlc <= CAN_MAX_DLC)))
391  dlc = len;
392 
393  buf[offset + 1] = '{';
394  buf[offset + 2] = hex_asc_upper[dlc];
395  buf[offset + 3] = '}';
396  } else {
397  buf[offset + 1] = '[';
398  buf[offset + 2] = len + '0';
399  buf[offset + 3] = ']';
400  }
401 
402  /* standard CAN frames may have RTR enabled */
403  if (cf->can_id & CAN_RTR_FLAG) {
404  sprintf(buf+offset+5, " remote request");
405  return;
406  }
407  } else {
408  buf[offset] = '[';
409  buf[offset + 1] = (len/10) + '0';
410  buf[offset + 2] = (len%10) + '0';
411  buf[offset + 3] = ']';
412  }
413  offset += 5;
414 
415  if (view & CANLIB_VIEW_BINARY) {
416  dlen = 9; /* _10101010 */
417  if (view & CANLIB_VIEW_SWAP) {
418  for (i = len - 1; i >= 0; i--) {
419  buf[offset++] = (i == len-1)?' ':SWAP_DELIMITER;
420  for (j = 7; j >= 0; j--)
421  buf[offset++] = (1<<j & cf->data[i])?'1':'0';
422  }
423  } else {
424  for (i = 0; i < len; i++) {
425  buf[offset++] = ' ';
426  for (j = 7; j >= 0; j--)
427  buf[offset++] = (1<<j & cf->data[i])?'1':'0';
428  }
429  }
430  } else {
431  dlen = 3; /* _AA */
432  if (view & CANLIB_VIEW_SWAP) {
433  for (i = len - 1; i >= 0; i--) {
434  if (i == len-1)
435  buf[offset++] = ' ';
436  else
437  buf[offset++] = SWAP_DELIMITER;
438 
439  put_hex_byte(buf + offset, cf->data[i]);
440  offset += 2;
441  }
442  } else {
443  for (i = 0; i < len; i++) {
444  buf[offset++] = ' ';
445  put_hex_byte(buf + offset, cf->data[i]);
446  offset += 2;
447  }
448  }
449  }
450 
451  buf[offset] = 0; /* terminate string */
452 
453  /*
454  * The ASCII & ERRORFRAME output is put at a fixed len behind the data.
455  * For now we support ASCII output only for payload length up to 8 bytes.
456  * Does it make sense to write 64 ASCII byte behind 64 ASCII HEX data on the console?
457  */
458  if (len > CAN_MAX_DLEN)
459  return;
460 
461  if (cf->can_id & CAN_ERR_FLAG)
462  sprintf(buf+offset, "%*s", dlen*(8-len)+13, "ERRORFRAME");
463  else if (view & CANLIB_VIEW_ASCII) {
464  j = dlen*(8-len)+4;
465  if (view & CANLIB_VIEW_SWAP) {
466  sprintf(buf+offset, "%*s", j, "`");
467  offset += j;
468  for (i = len - 1; i >= 0; i--)
469  if ((cf->data[i] > 0x1F) && (cf->data[i] < 0x7F))
470  buf[offset++] = cf->data[i];
471  else
472  buf[offset++] = '.';
473 
474  sprintf(buf+offset, "`");
475  } else {
476  sprintf(buf+offset, "%*s", j, "'");
477  offset += j;
478  for (i = 0; i < len; i++)
479  if ((cf->data[i] > 0x1F) && (cf->data[i] < 0x7F))
480  buf[offset++] = cf->data[i];
481  else
482  buf[offset++] = '.';
483 
484  sprintf(buf+offset, "'");
485  }
486  }
487 }

561 static int snprintf_error_data(char *buf, size_t len, uint8_t err,
562  const char **arr, int arr_len)
563 {
564  int i, n = 0, count = 0;
565 
566  if (!err || len <= 0)
567  return 0;
568 
569  for (i = 0; i < arr_len; i++) {
570  if (err & (1 << i)) {
571  int tmp_n = 0;
572  if (count){
573  /* Fix for potential buffer overflow https://lgtm.com/rules/1505913226124/ */
574  tmp_n = snprintf(buf + n, len - n, ",");
575  if (tmp_n < 0 || (size_t)tmp_n >= len - n){
576  return n;
577  }
578  n += tmp_n;
579  }
580  tmp_n = snprintf(buf + n, len - n, "%s", arr[i]);
581  if (tmp_n < 0 || (size_t)tmp_n >= len - n){
582  return n;
583  }
584  n += tmp_n;
585  count++;
586  }
587  }
588 
589  return n;
590 }

636 void snprintf_can_error_frame(char *buf, size_t len, const struct canfd_frame *cf,
637  const char* sep)
638 {
639  canid_t cls, mask;
640  int i, n = 0, classes = 0;
641  const char *defsep = ",";
642 
643  if (!(cf->can_id & CAN_ERR_FLAG))
644  return;
645 
646  cls = cf->can_id & CAN_EFF_MASK;
647  if (cls > (1 << ARRAY_SIZE(error_classes))) {
648  fprintf(stderr, "Error class %#x is invalid\n", cls);
649  return;
650  }
651 
652  if (!sep)
653  sep = defsep;
654 
655  for (i = 0; i < (int)ARRAY_SIZE(error_classes); i++) {
